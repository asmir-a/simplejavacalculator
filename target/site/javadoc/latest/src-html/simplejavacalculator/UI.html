206   @Override
207   public void actionPerformed(ActionEvent e) {
208      final Object source = e.getSource();
209      
210      for (int i = 0; i < 10; i++) {
211         if (source == but[i]) {
212            text.replaceSelection(buttonValue[i]);
213            return;
214         }
215      }
216      
217      if (source == butAdd) {
218         writer(calc.calculateBi(Calculator.BiOperatorModes.add, reader()));
219      }
220      
221      if (source == butMinus) {
222         writer(calc.calculateBi(Calculator.BiOperatorModes.minus, reader()));
223      }
224      
225      if (source == butMultiply) {
226         writer(calc.calculateBi(Calculator.BiOperatorModes.multiply,
227                                 reader()));
228      }
229      
230      if (source == butDivide) {
231         writer(calc
232                   .calculateBi(Calculator.BiOperatorModes.divide, reader()));
233      }
234      if (source == butxpowerofy) {
235         writer(calc
236                   .calculateBi(Calculator.BiOperatorModes.xpowerofy, reader()));
237      }
238      
239      if (source == butSquare) {
240         writer(calc.calculateMono(Calculator.MonoOperatorModes.square,
241                                   reader()));
242      }
243      
244      if (source == butSquareRoot) {
245         writer(calc.calculateMono(Calculator.MonoOperatorModes.squareRoot,
246                                   reader()));
247      }
248      
249      if (source == butOneDevidedBy) {
250         writer(calc.calculateMono(
251                                   Calculator.MonoOperatorModes.oneDevidedBy, reader()));
252      }
253      
254      if (source == butCos) {
255         writer(calc.calculateMono(Calculator.MonoOperatorModes.cos,
256                                   reader()));
257      }
258      
259      if (source == butSin) {
260         writer(calc.calculateMono(Calculator.MonoOperatorModes.sin,
261                                   reader()));
262      }
263      
264      if (source == butTan) {
265         writer(calc.calculateMono(Calculator.MonoOperatorModes.tan,
266                                   reader()));
267      }
268      if (source == butlog) {
269         writer(calc.calculateMono(Calculator.MonoOperatorModes.log,
270                                   reader()));
271      }
272      if (source == butrate) {
273         writer(calc.calculateMono(Calculator.MonoOperatorModes.rate,
274                                   reader()));
275      }
276      if(source == butabs){
277         writer(calc.calculateMono(Calculator.MonoOperatorModes.abs, reader()));
278      }
279      
280      if (source == butEqual) {
281         writer(calc.calculateEqual(reader()));
282      }
283      
284      if (source == butCancel) {
285         writer(calc.reset());
286      }
287      
288      if (source == butBinary) {
289         parsetoBinary();
290      }
291      
292      text.selectAll();
293   }
294   
295   private void parsetoBinary() {
296      try {
297         text.setText("" + Long.toBinaryString(Long.parseLong(text.getText())));
298      } catch (NumberFormatException ex) {
299         System.err.println("Error while parse to binary." + ex.toString());
300      }
301   }
302   
303   public Double reader() {
304      Double num;
305      String str;
306      str = text.getText();
307      num = Double.valueOf(str);
308      
309      return num;
310   }
311   
312   public void writer(final Double num) {
313      if (Double.isNaN(num)) {
314         text.setText("");
315      } else {
316         text.setText(Double.toString(num));
317      }
318   }
